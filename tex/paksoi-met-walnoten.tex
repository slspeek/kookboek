\begin{recipe}
[ %
    preparationtime = {\unit[20]{m}},
    portion = {\portion{2}},
    calory,
    source = {MissRiz op Smulweb}
]
{Paksoi met walnoten}

    \ingredients
    {%
      \unit[2]{kopjes} & witte rijst \\
      1 struik & paksoi \\
      1 & vleestomaat \\
      \unit[1]{tl} & sambal \\
      2 teentjes & knoflook \\
      6 & walnoten \\
      1 & bouillonblokje \\
      snufje & djinten \\
      snufje & ketoembar\\
      \unit[1]{tl}  & sesamzaad
    }

    \preparation
    {%
        \step Kook de rijst volgens de aanwijzingen op de verpakking.
        \step Paksoi in repen snijden,
              de stengels wat dunner dan de bladeren,
              knoflook fijn snijden,
              de tomaat in blokjes, walnoten kraken (niet hakken!).
              Verkruimel het bouillonblokje met een oud mes op een snijplank.
        \step Knoflook 1 minuut fruiten, dan 2 minuten met de sambal erbij.
              Dan de tomaat, bouillonpoeder en kruiden erbij doen. Even bakken.
              Dan de paksoi erbij doen, 2-3 minuten roerbakken.
              Als laatste de walnoten erdoor scheppen.
              Niet te lang in de pan laten, want de paksoi moet geen slappe hap worden.
      }

\end{recipe}
\label{rec:paksoi-met-walnoten}
