\begin{recipe}
[ %
    preparationtime = {\unit[15]{min}},
    portion = {\portion{3}},
    source = {\href{https://www.ah.nl/allerhande/recept/R-R1197241/bloemkoolsaus}{Allerhande}}
]
{Kaassaus}

    \ingredients
    {%
        \unit[150]{ml} & melk \\
        \unit[1]{el} & maizena \\
        \unit[\nicefrac{1}{2}]{tl} & nootmuskaat \\
        \unit[120]{g} & geraspte kaas \\
	     & zout \\
         & peper 
    }

    \preparation
    {%
	    \step Meng de maizena met de melk terwijl alles nog koel is.
	    \step Breng de melk met de kaas aan de kook en laat het twee minuten doorkoken.
        \step Breng op smaak met peper, zout en de nootmuskaat.
    }

    \suggestion
    {%
        Lekker bij bloemkool, witlof en broccoli.
    }
    
\end{recipe}
