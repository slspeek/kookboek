\begin{recipe}
[ %
    preparationtime = {\unit[30]{min}},
    portion = {\portion{2}},
]
{Pasta met mozzarella en tomaat}

    \ingredients
    {%
        \unit[250]{g} & pasta\\
        \unit[1]{bol} & mozzarella\\
        & Parmezaanse kaas\\
        4 & tomaten\\
        \unit[2]{teentjes} & knoflook\\
        1 & ui\\
        & oregano\\
        & olijfolie\\
        & zout\\
        & peper
    }
	
    \preparation
    {%
	    \step Kook de pasta volgens de aanwijzingen op de verpakking.
        \step Snijd de mozzarella in blokjes en de tomaten in stukjes. Snijd de knoflook fijn.
        \step Verhit een scheut olijfolie in een pan en fruit de ui en knoflook.
        \step Voeg de tomaten toe en bak ze 5 minuten mee. Breng op smaak met oregano, zout en peper.
        \step Voeg de mozzarella toe en laat deze smelten in de saus. Niet roeren, want dan
        gaat de mozzarella klonteren.
        \step Schep de pasta op de borden en schep de saus eroverheen.
        \step Serveer met Parmezaanse kaas.
    }
    
\end{recipe}
