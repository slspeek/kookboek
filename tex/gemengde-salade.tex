\begin{recipe}
[ %
    preparationtime = {\unit[10]{m}},
    portion = {\portion{2}},
]
{Gemengde salade}

    \ingredients
    {%
         \unit[\nicefrac{1}{2}]{zakje} & sla \\
          1 & tomaat \\
         \nicefrac{1}{3} & komkommer \\
    }

    \preparation
    {%
        \step Snijd de tomaat in blokjes.
        \step Schil eventueel de komkommer. Snijd de komkommer in blokjes.
        \step Meng de sla, tomaat en komkommer in een schaal.
        \step Breng op smaak met peper en zout. (Let op dan geen peper en zout in de vinaigrette doen.)
        \step Maak een \hyperref[rec:vinaigrette]{vinaigrette} en serveer deze bij de salade.
      }

    \suggestion
    {
      Je kunt ook een half fijngesnipperd uitje toevoegen om het pittiger te maken.
    }
    
    \suggestion
    {
      Eventueel kun je een hardgekookt eitje toevoegen aan de salade.
    }

    \suggestion
    {
      Je kunt ook andere groenten toevoegen zoals paprika, wortel of radijsjes.
    }
   
\end{recipe}
\label{rec:gemengde_salade}
