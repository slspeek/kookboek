\begin{recipe}
[ %
    preparationtime = {\unit[10]{m}},
    portion = {\portion{2}},
]
{Gemengde salade}

    \ingredients
    {%
         \unit[\nicefrac{1}{2}]{zakje} & sla \\
          1 & tomaat \\
         \nicefrac{1}{3} & komkommer \\
    }

    \preparation
    {%
        \step Snijd de tomaat in blokjes.
        \step Schil eventueel de komkommer. Snijd de komkommer in blokjes.
        \step Meng de sla, tomaat en komkommer in een schaal.
        \step Breng op smaak met peper en zout. (Let op dan geen peper en zout in de vinaigrette doen.)
        \step Maak een \linkto{rec:vinaigrette}{vinaigrette} en serveer deze bij de salade.
      }

    \suggestion
    {
      Je kunt ook nog: 
      \begin{itemize}
        \item half fijngesnipperd uitje 
        \item een hardgekookt eitje
        \item paprika, wortel of radijsjes
        \item peterselie, bieslook of basilicum
        \item een handje pijnboompitten of zonnebloempitten
      \end{itemize}
      toevoegen aan de salade.
    }
   
\end{recipe}
\label{rec:gemengde_salade}
