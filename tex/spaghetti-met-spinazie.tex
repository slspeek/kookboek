\begin{recipe}
[ %
    preparationtime = {\unit[45]{min}},
    portion = {\portion{2}},
]
{Spaghetti met spinazie}

    \ingredients
    {%
        zak & verse spinazie (\unit[500]{g} ) \\
        1 & ui \\
        \unit[3]{teentjes} & knoflook \\
        \unit[250]{g} & pasta \\
        \unit[\nicefrac{3}{4}]{beker} & zure room \\
        \unit[1]{tl} & Italiaanse keukenkruiden \\
        verpakking & zeeuws-/ontbijtspek \\
        & olijfolie \\
        & parmasaanse kaas \\
	    & zout \\
        & peper \\
    }

    \preparation
    {%
	    \step Snipper de ui en snijd de knoflook fijn.
        \step Snijd de spinazie. Gebruik een schone afwasteil om de gesneden 
        spinazie in te doen.
        \step Bak in een koekepan het spek rustig uit in 20 tot 25 minuten.
        \step Fruit in een hapjespan de ui en de knoflook in de olijfolie.
        \step Voeg stap voor stap de gesneden spinazie toe aan de hapjespan.
        \step Voeg de kruiden aan het spinaziemengsel toe.
        \step Kook de pasta. Laat ondertussen de spinazie een beetje inkoken.
        \step Voeg het spek en de zure room aan het spinaziemengsel toe.
	    \step Rasp de parmasaanse kaas.
    }
    
\end{recipe}
