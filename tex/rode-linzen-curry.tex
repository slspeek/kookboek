\begin{recipe}
[ %
    preparationtime = {\unit[1\nicefrac{1}{2}]{uur}},
    portion = {\portion{4}},
    source = {\href{https://vegafoodness.nl/linzen-curry/}{Vegafoodness}}
]
{Rode linzen curry}

    \ingredients
    {%
     \unit[200]{g} & gedroogde rode linzen \\
     1 & (punt) paprika \\
     kleine & ui \\
     1 blik & tomaten blokjes (\unit[400]{g}) \\
     1 blik & kokosmelk (\unit[400]{g}) \\
     \unit[1]{el} & olie \\
     \unit[100]{ml} & water \\
     1 & rode peper \\
     \unit[2]{tenen} & knoflook \\
     \unit[1]{cm} & gember \\
     \unit[2]{tl} & paprikapoeder \\
     \unit[2]{tl} & komijn \\
     \unit[\nicefrac{1}{2}]{tl} & kurkuma \\
     \unit[\nicefrac{1}{2}]{tl} & zout \\
     \unit[1]{tl} & gedroogde tijm \\
     \unit[1]{tl} & garam massala \\
     4 & naan broodjes \\
     2 & limoenen \\
     zakje & verse koriander \\
     & witte rijst \\
	}

    \preparation
    {%
	    \step Was de rode linzen en laat ze even uitlekken.
	    \step Snipper de ui en snijd de knoflook fijn. Verwijder de zaden van de rode paprika
         en snijd deze in blokjes. Verwijder de schil van de gember en hak deze heel fijn en
         als laatste snijd je ook de rode peper fijn.
        \step Verhit een lepel olie in een pan en fruit de ui en knoflook hierin. Voeg 
         vervolgens de gember, rode peper en punt paprika toe. Bak dit 2-3 minuten.
        \step Voeg nu alle kruiden toe en giet vervolgens de kokosmelk, tomaten blokjes en
         het water hieraan toe. Roer alles goed door elkaar en voeg daarna de linzen toe.
        \step Breng het geheel zachtjes aan de kook en laat voor zo'n 45 minuten op het vuur
         staan met de deksel op de pan. Kook ondertussen alvast de rijst gaar.
        \step Verwarm de naan broodjes.
        \step Wanneer de linzen gaar gekookt zijn haal je de pan van het vuur en verdeel je
         de rijst en curry over 4 borden. Besprenkel met limoen en strooi er
         wat verse koriander overheen.
    }
    
\end{recipe}
