\begin{recipe}
[ %
    preparationtime = {\unit[8]{m}},
    portion = {\portion{4 - 6}},
]
{Vinaigrette}

    \ingredients
    {%
         \unit[3]{el} & olijfolie \\
         \unit[2]{tl} & mosterd \\
         \unit[1]{el} & azijn \\
         \unit[1]{el} & honing of gembersiroop \\
         \unit[4]{tenen} & knoflook \\
         & zout en peper\\
    }

    \preparation
    {%
        \step Pers de knoflooktenen uit en doe deze in een kommetje.
        \step Voeg de overige ingredi\"{e}nten toe. En roer
        het goed.
        \step Voeg zout en peper naar smaak toe.
    }

\end{recipe}
